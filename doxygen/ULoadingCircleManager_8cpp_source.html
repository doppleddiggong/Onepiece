29void ULoadingCircleManager::EnsureWidgetForWorld(UWorld* World)
30{
31 if (World == nullptr || !World->IsGameWorld())
32 return;
33
34 // 위젯이 유효하고 같은 월드이며, 뷰포트에 추가되어 있는지 확인
35 const bool bIsValid = IsValid(CircleWidget);
36 const bool bSameWorld = bIsValid && CircleWidget->GetWorld() == World;
37 const bool bInViewport = bIsValid && CircleWidget->IsInViewport();
38
39 if (bIsValid && bSameWorld && bInViewport)
40 return;
41
42 // 기존 위젯이 있으면 정리
43 if (CircleWidget)
44 {
45 if (CircleWidget->IsInViewport())
46 CircleWidget->RemoveFromParent();
47
48 CircleWidget = nullptr;
49 }
50
51 ULocalPlayer* LocalPlayer = GetLocalPlayer();
52 if (LocalPlayer == nullptr)
53 return;
54
55 // 멀티플레이 대응: PlayerController 사용
56 APlayerController* PC = LocalPlayer->GetPlayerController(World);
57 if (PC == nullptr)
58 return;
59
60 if (!CircleWidgetClass)
61 return;
62
63 if (ULoadginCircle* NewWidget = CreateWidget<ULoadginCircle>(PC, CircleWidgetClass))
64 {
65 // Game Viewport에 추가하여 레벨 전환 시에도 유지
66 NewWidget->AddToGameViewport(GameLayer::LoadingCircle);
67 CircleWidget = NewWidget;
68
69 // 위젯 재생성 시 현재 카운트에 따라 표시 상태 복원
70 if (LoadingCount > 0)
71 {
72 CircleWidget->Show();
73 }
74 else
75 {
76 CircleWidget->Hide();
77 }
78 }
79}

81void ULoadingCircleManager::Show()
82{
83 LoadingCount++;
84
85 PRINTLOG(TEXT("[LoadingCircleManager] Show - Count: %d → %d"), LoadingCount - 1, LoadingCount);
86
87 if (UWorld* World = GetWorld())
88 {
89 EnsureWidgetForWorld(World);
90
91 if (CircleWidget && LoadingCount > 0)
92 {
93 CircleWidget->Show();
94 }
95 else if (!CircleWidget)
96 {
97 PRINTLOG(TEXT("[LoadingCircleManager] Show FAILED - Widget is nullptr!"));
98 }
99 }
100}

102void ULoadingCircleManager::Hide()
103{
104 const int32 OldCount = LoadingCount;
105 LoadingCount = FMath::Max(0, LoadingCount - 1);
106
107 PRINTLOG(TEXT("[LoadingCircleManager] Hide - Count: %d → %d"), OldCount, LoadingCount);
108
109 if (OldCount == 0)
110 {
111 PRINTLOG(TEXT("[LoadingCircleManager] WARNING: Hide called but count was already 0!"));
112 }
113
114 if (UWorld* World = GetWorld())
115 {
116 EnsureWidgetForWorld(World);
117
118 if (CircleWidget)
119 {
120 if (LoadingCount == 0)
121 {
122 CircleWidget->Hide();
123 }
124 }
125 else
126 {
127 PRINTLOG(TEXT("[LoadingCircleManager] Hide FAILED - Widget is nullptr!"));
128 }
129 }
