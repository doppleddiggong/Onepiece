27void UDialogManager::EnsureWidgetForWorld(UWorld* World)
28{
29 if (World == nullptr || !World->IsGameWorld())
30 return;
31
32 // 위젯이 유효하고 같은 월드이며, 뷰포트에 추가되어 있는지 확인
33 if (IsValid(DialogWidget) &&
34 DialogWidget->GetWorld() == World &&
35 DialogWidget->IsInViewport())
36 {
37 return;
38 }
39
40 // 기존 위젯이 있으면 정리
41 if (DialogWidget)
42 {
43 if (DialogWidget->IsInViewport())
44 DialogWidget->RemoveFromParent();
45
46 DialogWidget = nullptr;
47 }
48
49 ULocalPlayer* LocalPlayer = GetLocalPlayer();
50 if (LocalPlayer == nullptr)
51 return;
52
53 APlayerController* PC = LocalPlayer->GetPlayerController(World);
54 if (PC == nullptr)
55 return;
56
57 if (!DialogWidgetClass)
58 return;
59
60 UDialogWidget* NewWidget = CreateWidget<UDialogWidget>(PC, DialogWidgetClass);
61 if (!NewWidget)
62 return;
63
64 NewWidget->AddToViewport(GameLayer::ToastManager);
65 DialogWidget = NewWidget;
66}
