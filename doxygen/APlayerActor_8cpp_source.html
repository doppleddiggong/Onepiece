116void APlayerActor::Cmd_Move_Implementation(const FVector2D& Axis)
117{
118 if ( !Controller)
119 {
120 return;
121 }
122
123 // Use the controller's rotation to determine movement direction.
124 const FRotator ControlRotation = Controller->GetControlRotation();
125 const EMovementMode CurrentMovementMode = GetCharacterMovement()->MovementMode;
126
127 if (CurrentMovementMode == MOVE_Walking || CurrentMovementMode == MOVE_Falling)
128 {
129 // For ground movement, only use the Yaw rotation to prevent pitching into the ground.
130 const FRotator YawRotation(0.0f, ControlRotation.Yaw, 0.0f);
131
132 // Calculate forward and right vectors based on the Yaw rotation.
133 const FVector ForwardDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::X);
134 const FVector RightDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::Y);
135
136 AddMovementInput(ForwardDirection, Axis.Y);
137 AddMovementInput(RightDirection, Axis.X);
138 }
139 else if (CurrentMovementMode == MOVE_Flying)
140 {
141 // For flying, use the full 3D rotation of the controller.
142 const FRotator FullRotation(ControlRotation.Pitch, ControlRotation.Yaw, 0.0f);
143
144 const FVector ForwardDirection = FRotationMatrix(FullRotation).GetUnitAxis(EAxis::X);
145 const FVector RightDirection = FRotationMatrix(FullRotation).GetUnitAxis(EAxis::Y);
146
147 AddMovementInput(ForwardDirection, Axis.Y);
148 AddMovementInput(RightDirection, Axis.X);
149 }
150}
