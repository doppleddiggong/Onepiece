33bool FHttpMultipartFormData::AddFile(const FString& FieldName, const FString& FilePath, const FString& InMimeType)
34{
35 TArray<uint8> FileBytes;
36 if (!FFileHelper::LoadFileToArray(FileBytes, *FilePath))
37 {
38 UE_LOG(LogTemp, Warning, TEXT("[HttpMultipartFormData] Failed to load file: %s"), *FilePath);
39 return false;
40 }
41
42 FString FileNameOnly = FPaths::GetCleanFilename(FilePath);
43 FString Mime = InMimeType.IsEmpty() ? DetectMimeFromExtension(FileNameOnly) : InMimeType;
44
45 FileParts.Add({ FieldName, FileNameOnly, Mime, MoveTemp(FileBytes) });
46 return true;
47}

63void FHttpMultipartFormData::BuildBody()
64{
65 BuiltBody.Reset();
66
67 auto Add = [](TArray<uint8>& Dest, const FString& S)
68 {
69 AppendUtf8(Dest, S);
70 };
71
72 const FString LineEnd = TEXT("\r\n");
73 const FString BoundaryLine = TEXT("--") + Boundary;
74
75 // 텍스트 파트
76 for (const FTextPart& P : TextParts)
77 {
78 Add(BuiltBody, BoundaryLine + LineEnd);
79 Add(BuiltBody, FString::Printf(
80 TEXT("Content-Disposition: form-data; name=\"%s\"%s"),
81 *P.Name, *LineEnd));
82 Add(BuiltBody, LineEnd); // 헤더와 값 사이의 빈 줄
83 Add(BuiltBody, P.Value + LineEnd);
84 }
85
86 // 파일 파트
87 for (const FFilePart& P : FileParts)
88 {
89 Add(BuiltBody, BoundaryLine + LineEnd);
90 Add(BuiltBody, FString::Printf(
91 TEXT("Content-Disposition: form-data; name=\"%s\"; filename=\"%s\"%s"),
92 *P.Name, *P.FileName, *LineEnd));
93 Add(BuiltBody, FString::Printf(
94 TEXT("Content-Type: %s%s"),
95 *P.MimeType, *LineEnd));
96 Add(BuiltBody, LineEnd); // 바이너리 데이터 앞의 빈 줄
97
98 // 바이너리 데이터
99 BuiltBody.Append(P.Data);
100
101 // 파일 내용 뒤의 CRLF
102 AppendUtf8(BuiltBody, LineEnd);
103 }
104
105 // 닫는 경계
106 Add(BuiltBody, BoundaryLine + TEXT("--") + LineEnd);
107}

115FString FHttpMultipartFormData::DetectMimeFromExtension(const FString& FileName)
116{
117 const FString Ext = FPaths::GetExtension(FileName, /*bIncludeDot*/ false).ToLower();
118
119 // 일반적인 오디오 타입
120 if (Ext == TEXT("mp3")) return TEXT("audio/mpeg");
121 if (Ext == TEXT("wav")) return TEXT("audio/wav");
122 if (Ext == TEXT("ogg")) return TEXT("audio/ogg");
123 if (Ext == TEXT("flac")) return TEXT("audio/flac");
124 if (Ext == TEXT("m4a")) return TEXT("audio/mp4");
125
126 // 대체 타입
127 if (Ext == TEXT("txt")) return TEXT("text/plain");
128 if (Ext == TEXT("json")) return TEXT("application/json");
129
130 // 기본 타입
131 return TEXT("application/octet-stream");
132}
