70void UHttpNetworkSystem::RequestHealth( FResponseHealthDelegate InDelegate )
71{
72 auto HttpRequest = FHttpModule::Get().CreateRequest();
73
74 HttpRequest->SetVerb(NETWORK_GET);
75 HttpRequest->SetURL(NetworkConfig::GetFullUrl(RequestAPI::Health));
76 HttpRequest->SetHeader(TEXT("Content-Type"), TEXT("application/json"));
77
78 LogNetwork(ENetworkLogType::Get, *HttpRequest->GetURL());
79
80 HttpRequest->OnProcessRequestComplete().BindLambda(
81 [WeakThis = TWeakObjectPtr<UHttpNetworkSystem>(this), InDelegate](FHttpRequestPtr Req, FHttpResponsePtr ResPtr, bool bWasSuccessful)
82 {
83 if (!WeakThis.IsValid() || IsEngineExitRequested())
84 return;
85
86 WeakThis->AddNetworkWaitCount(-1);
87 FResponseHealth ResponseData;
88 ResponseData.SetFromHttpResponse(ResPtr);
89
90 InDelegate.ExecuteIfBound(ResponseData, bWasSuccessful);
91 });
92
93 AddNetworkWaitCount(1);
94 HttpRequest->ProcessRequest();
95}
