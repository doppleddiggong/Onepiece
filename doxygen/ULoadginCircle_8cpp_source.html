55void ULoadginCircle::UpdateLoadingSpinner(float DeltaTime)
56{
57 if (!LoadingSpinner || !LoadingSpinner->IsVisible())
58 return;
59
60 const float NewAngle = LoadingSpinner->GetRenderTransformAngle() + (SpinnerRotationSpeed * DeltaTime);
61 LoadingSpinner->SetRenderTransformAngle(NewAngle);
62}

64void ULoadginCircle::UpdateVisibility(bool bShouldShow)
65{
66 if (!RootOverlay)
67 {
68 PRINTLOG(TEXT("[LoadingCircle] UpdateVisibility: RootOverlay is nullptr!"));
69 return;
70 }
71
72 // PRINTLOG(TEXT("[LoadingCircle] UpdateVisibility - Visible: %s"),
73 // bShouldShow ? TEXT("TRUE") : TEXT("FALSE"));
74
75 if (bShouldShow)
76 {
77 // 로딩 중일 때: 보이고, 터치 차단
78 SetVisibility(ESlateVisibility::Visible);
79 RootOverlay->SetVisibility(ESlateVisibility::Visible);
80 }
81 else
82 {
83 // 로딩 완료 시: 안 보이지만 Hit Test 완전히 무시 (터치 이벤트 통과)
84 SetVisibility(ESlateVisibility::HitTestInvisible);
85 RootOverlay->SetVisibility(ESlateVisibility::Collapsed);
86 }
87}
